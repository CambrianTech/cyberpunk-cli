1. Full Terminal Interface

Create a complete cyberpunk terminal for your web application:

<!-- HTML --> <link rel="stylesheet" href="cyberpunk-web.css"> <script src="cyberpunk-web.js"></script> <div id="my-terminal"></div> <script> const terminal = new CyberpunkTerminal('my-terminal', { theme: 'loki', themeChanging: true }); terminal.setTitle('My Cyberpunk App') .addMenuItem('deploy', 'Deploy to Production', 'Deploy your application', () => { terminal.log('üöÄ Deploying to production...'); terminal.log('‚úÖ Deployment successful!'); }) .addMenuItem('monitor', 'System Monitoring', 'View system metrics', () => { terminal.log('üìä System Status: All systems operational'); terminal.log('üíæ Memory: 2.1GB / 8GB'); terminal.log('‚ö° CPU: 45% utilization'); }) .addSeparator('Database') .addMenuItem('backup', 'Database Backup', 'Create database backup', () => { terminal.log('üíæ Creating database backup...'); setTimeout(() => terminal.log('‚úÖ Backup complete!'), 2000); }); </script>

2. Simple Console Widget

Embed a cyberpunk console anywhere on your site:

<div id="my-console" style="max-width: 600px;"></div> <script> const console = new CyberpunkConsole('my-console', { theme: 'matrix', title: 'Development Console' }); console.log('System initialized'); console.log('Ready for commands'); </script>

3. Game Interface

Perfect for game development or interactive experiences:

const gameTerminal = new CyberpunkTerminal('game-interface', { theme: 'tron', themeChanging: false }); gameTerminal.setTitle('Cyberpunk Game Console') .addMenuItem('newgame', 'New Game', 'Start a new cyberpunk adventure', startNewGame) .addMenuItem('loadgame', 'Load Game', 'Continue your journey', loadGame) .addMenuItem('settings', 'Settings', 'Configure game options', showSettings) .addSeparator('Debug') .addMenuItem('godmode', 'God Mode', 'Toggle invincibility', toggleGodMode) .addMenuItem('teleport', 'Teleport', 'Teleport to location', showTeleportMenu);

4. DevOps Dashboard

Create professional DevOps tools with cyberpunk aesthetics:

const devopsTerminal = new CyberpunkTerminal('devops-dashboard', { theme: 'fallout' }); devopsTerminal.setTitle('DevOps Command Center') .addMenuItem('status', 'System Status', 'Check all services', checkSystemStatus) .addMenuItem('deploy', 'Deploy', 'Deploy to environments', showDeployMenu) .addMenuItem('logs', 'View Logs', 'Application and system logs', viewLogs) .addSeparator('Monitoring') .addMenuItem('metrics', 'Metrics', 'Performance metrics', showMetrics) .addMenuItem('alerts', 'Alerts', 'Active alerts and warnings', showAlerts);

5. Theme Customization

All themes from the Python package are available:

// Loki Theme - Green/Gold Nordic terminal.setTheme('loki'); // Matrix Theme - Digital Rain terminal.setTheme('matrix'); // Fallout Theme - Vault-Tec Amber terminal.setTheme('fallout'); // Tron Theme - Neon Blue Grid terminal.setTheme('tron'); // Enable/disable theme switching const terminal = new CyberpunkTerminal('container', { theme: 'loki', themeChanging: true // Allow Ctrl+T switching });
